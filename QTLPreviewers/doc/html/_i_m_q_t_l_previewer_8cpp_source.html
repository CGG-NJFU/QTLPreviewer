222 double calculateLOD(const vector<vector<double> > gp, const int sampleSize, const vector<EXPData> expData,
223  const EXPData u0, const EXPData s0,
224  const EXPData u1, const EXPData u2, const EXPData u3, const EXPData s1,
225  const bool ifPrintLog = VERBOSE_MODE) {
226  double sum1 = 0, sum2 = 0;
227  //cout <<"u0=" <<u0 <<endl <<"s0=" <<s0 <<endl;
228  for (int i = 0; i < sampleSize; i++) {
229  sum1 += log10(f(expData[i], u0, s0));
230  sum2 += log10(
231  gp[i][0] * f(expData[i], u1, s1)
232  + gp[i][1] * f(expData[i], u2, s1)
233  + gp[i][2] * f(expData[i], u3, s1));
234  }
235 
236  // LOD=log(L0/L)=logL0-LogL
237  // 连乘Log改为加法，除法改减法
238  double lod;
239  lod = sum2 - sum1;
240 
241  if (ifPrintLog) {
242  cout << "sum1=" << sum1 << "\tsum2=" << sum2 << "\tLOD=" << lod << endl;
243  }
244  return lod;
245 }

260 void EMCalculate(const vector<vector<double> >& gp, const int sampleSize, const vector<EXPData> expData,
261  const EXPData u0, const EXPData s0, EXPData* u1, EXPData* u2, EXPData* u3, EXPData* s1,
262  const bool ifPrintLog = VERBOSE_MODE) {
263  // u1, u2, u3, sigma2 见113
264  EXPData u10 = u0;
265  EXPData u20 = u0;
266  EXPData u30 = u0;
267  EXPData s10 = s0; //*s1+1;
268 
269  // 收敛精度要求
270  int step = 1;
271  if (ifPrintLog) {
272  cout << "=======EMSteps======" << endl;
273  cout << "[step]\tu1\tu2\tu3\ts1\n";
274  }
275  while (fabs(u10 - *u1) > ACCURACY_REQ || fabs(u20 - *u2) > ACCURACY_REQ
276  || fabs(u30 - *u3) > ACCURACY_REQ || fabs(s10 - *s1) > ACCURACY_REQ) {
277  double py1 = 0, py10 = 0;
278  double py2 = 0, py20 = 0;
279  double py3 = 0, py30 = 0;
280  double py4 = 0, py5 = 0, py6 = 0;
281 
282  int i;
283  for (i = 0; i < sampleSize; i++) {
284  double gpi1f = gp[i][0] * f(expData[i], u10, s10);
285  double gpi2f = gp[i][1] * f(expData[i], u20, s10);
286  double gpi3f = gp[i][2] * f(expData[i], u30, s10);
287  double gpi = gpi1f + gpi2f + gpi3f;
288 
289  py1 += gpi1f / gpi; //PI1的值
290  py2 += gpi2f / gpi; //PI2的值
291  py3 += gpi3f / gpi; //PI3的值
292 
293  py10 += gpi1f / gpi * expData[i]; //PI1和PI1*YI的和
294  py20 += gpi2f / gpi * expData[i]; //PI2和PI2*YI的和
295  py30 += gpi3f / gpi * expData[i]; //PI3和PI3*YI的和
296 
297  py4 += gpi1f / gpi * (expData[i] - u10) * (expData[i] - u10);
298  py5 += gpi2f / gpi * (expData[i] - u20) * (expData[i] - u20);
299  py6 += gpi3f / gpi * (expData[i] - u30) * (expData[i] - u30);
300  }
301 
302  *u1 = u10;
303  u10 = py10 / py1;
304  *u2 = u20;
305  u20 = py20 / py2;
306  *u3 = u30;
307  u30 = py30 / py3;
308  *s1 = s10;
309  s10 = (py4 + py5 + py6) / sampleSize;
310 
311  if (ifPrintLog) {
312  cout << "[" << step++ << "]\t" << *u1 << "\t" << *u2 << "\t" << *u3
313  << "\t" << *s1 << endl;
314  }
315  } //end-while
316 
317  //输出终值
318  *u1 = u10;
319  *u2 = u20;
320  *u3 = u30;
321  *s1 = s10;
322  if (ifPrintLog) {
323  cout << endl << "Finals:" << endl;
324  cout << "u1=" << *u1 << "\tu2=" << *u2 << "\tu3=" << *u3 << "\ts1="
325  << *s1 << endl;
326  }
327 }

338 double calcCross(const bitset<GENE_CP_BIT> geneBit,
339  const double r1, const double r2, const double r=0, const double zeta=1) {
340  if (geneBit.size()!=GENE_CP_BIT) return 0;
341 
342  double r12 = r1*r2*zeta;
343  double r0 = r;
344  if (r==0) {
345  r0 = r1+r2-2*r12;
346  }
347 
348  // 配子产生的概率，见书105页
349  switch(geneBit.to_ulong()) {
350  case 0: //M1QM2 000
351  case 7: //m1qm2 111
352  return (1.0-r1-r2+r12);// `=(1-r1)(1-r2);
353  case 1: //M1Qm2 001
354  case 6: //m1qM2 110
355  return (r2-r12);// `=r2(1-r1);
356  case 4: //M1qm2 100
357  case 3: //m1QM2 011
358  return (r1-r12);// `=r1(1-r2);
359  case 5: //M1qM2 101
360  case 2: //m1Qm2 010
361  return r12;// `=r1*r2;
362  }
363 
364  return 0;
365 }

395 int calcGeneCP(vector<string>& geneMatrix, vector<double>& rMatrix,
396  const string f, const string m, const string q,
397  const double r1, const double r2, const double r=0, const double zeta=1,
398  const bool ifReorder=true, const bool ifPrintLog = VERBOSE_MODE) {
399  for (unsigned int i=0; i<geneMatrix.size(); i++) {
400  bitset<4> bs_fm(i/GENE_CP_CUT); //前4位为亲本基因型
401  bitset<2> bs_qtl(i%GENE_CP_CUT); //后2位为QTL基因型
402 
403  int fGene = bs_fm[3]*4 + bs_qtl[1]*2 + bs_fm[1]; //父本基因型
404  int mGene = bs_fm[2]*4 + bs_qtl[0]*2 + bs_fm[0]; //母本基因型
405 
406  bitset<3> bs_up(fGene);
407  bitset<3> bs_down(mGene);
408 
409  rMatrix[i] = calcCross(bs_up, r1, r2, r, zeta) * calcCross(bs_down, r1, r2, r, zeta);
410 
411  geneMatrix[i] =
412  (i&0x20?f.substr(1,1):f.substr(0,1)) +
413  (i&0x10?f.substr(3,1):f.substr(2,1)) +
414  (i&0x08?m.substr(1,1):m.substr(0,1)) +
415  (i&0x04?m.substr(3,1):m.substr(2,1)) +
416  (i&0x02?q.substr(1,1):q.substr(0,1)) +
417  (i&0x01?q.substr(1,1):q.substr(0,1));
418  }
419 
420  if (ifReorder) {
421  reorderGene(geneMatrix);
422  }
423 
424  if (ifPrintLog) {
425  printGeneCP(geneMatrix, rMatrix);
426  }
427  return 1;
428 }

447 double intervalQTL(const int currentTrait, const EXPData u0, const EXPData s0, const double length,
448  const double startPoint, const vector<vector<string> >& mk, const vector<EXPData> expData, const int sampleSize,
449  const string f, const string m, const string q, const bool ifUseShortGeneData=false,
450  const bool ifPrintLog = VERBOSE_MODE) {
451  double r = d2r(length);
452  double r1 = d2r(startPoint);
453  double r2 = d2r(length - startPoint);
454 
455  if (ifPrintLog) {
456  cout << "r=" << r << "\tr1=" << r1 << "\tr2=" << r2 << "\td1="
457  << startPoint << "\td2=" << (length - startPoint) << "\td="
458  << length << endl;
459  }
460 
461  //计算基因型条件概率分布全数据，相当于生成书107页表
462  vector<double> rMatrix(GENE_CP_SIZE);
463  vector<string> geneMatrix(GENE_CP_SIZE);
464  calcGeneCP(geneMatrix, rMatrix, f, m, q, r1, r2, r, 1, true, ifPrintLog);
465 
466  //统计各类型分布比率
467  vector<vector<double> > gp = vector<vector<double> >(sampleSize, vector<double>(3));
468  calculateGP(gp, mk, q, sampleSize, currentTrait, currentTrait + 1, geneMatrix, rMatrix, ifUseShortGeneData, ifPrintLog);
469 
470  double u1 = 0, u2 = 0, u3 = 0, s1 = 0;
471  // EM算法
472  EMCalculate(gp, sampleSize, expData, u0, s0, &u1, &u2, &u3, &s1, ifPrintLog);
473 
474  return calculateLOD(gp, sampleSize, expData, u0, s0, u1, u2, u3, s1, ifPrintLog);
475 }
