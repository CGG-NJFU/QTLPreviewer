204 void calculatePij(double p[9][3], double r, double r1, double r2,
205  int ifPrintLog = VERBOSE_MODE) {
206  p[0][0] = (1 - r1) * (1 - r1) * (1 - r2) * (1 - r2) / ((1 - r) * (1 - r));
207  p[0][1] = 2 * r1 * r2 * (1 - r1) * (1 - r2) / ((1 - r) * (1 - r));
208  p[0][2] = r1 * r1 * r2 * r2 / ((1 - r) * (1 - r));
209  p[1][0] = (1 - r1) * (1 - r1) * r2 * (1 - r2) / (r * (1 - r));
210  p[1][1] = r1 * (1 - r1) * (1 - 2 * r2 + 2 * r2 * r2) / (r * (1 - r));
211  p[1][2] = r1 * r1 * r2 * (1 - r2) / (r * (1 - r));
212  p[2][0] = (1 - r1) * (1 - r1) * r2 * r2 / (r * r);
213  p[2][1] = 2 * r1 * r2 * (1 - r1) * (1 - r2) / (r * r);
214  p[2][2] = r1 * r1 * (1 - r2) * (1 - r2) / (r * r);
215  p[3][0] = r1 * (1 - r1) * (1 - r2) * (1 - r2) / (r * (1 - r));
216  p[3][1] = (1 - 2 * r1 + 2 * r1 * r1) * r2 * (1 - r2) / (r * (1 - r));
217  p[3][2] = r1 * (1 - r1) * r2 * r2 / (r * (1 - r));
218  p[4][0] = 2 * r1 * r2 * (1 - r1) * (1 - r2) / (1 - 2 * r + 2 * r * r);
219  p[4][1] = (1 - 2 * r1 + 2 * r1 * r1) * (1 - 2 * r2 + 2 * r2 * r2)
220  / (1 - 2 * r + 2 * r * r);
221  p[4][2] = p[4][0];
222  p[5][0] = p[3][2];
223  p[5][1] = p[3][1];
224  p[5][2] = p[3][0];
225  p[6][0] = p[2][2];
226  p[6][1] = p[2][1];
227  p[6][2] = p[2][0];
228  p[7][0] = p[1][2];
229  p[7][1] = p[1][1];
230  p[7][2] = p[1][0];
231  p[8][0] = p[0][2];
232  p[8][1] = p[0][1];
233  p[8][2] = p[0][0];
234 
235  if (ifPrintLog) {
236  for (int i = 0; i < 9; i++) {
237  double sum = 0;
238  for (int j = 0; j < 3; j++) {
239  sum += p[i][j];
240  cout << p[i][j] << "\t";
241  }
242  cout << sum << "=1.0" << endl;
243  }
244  }

322 double calculateLOD(double** _gp, int sampleSize, double* y, double s0,
323  double s1, double u0, double u1, double u2, double u3, int ifPrintLog =
324  VERBOSE_MODE) {

327  //cout <<"u0=" <<u0 <<endl <<"s0=" <<s0 <<endl;
328  for (int i = 0; i < sampleSize; i++) {
329  sum1 += log10(f(y[i], u0, s0));

331  *(gp + i * sampleSize + 0) * f(y[i], u1, s1)

360 void EMCalculate(double* y, double** _gp, const int sampleSize, double u0,
361  double s0, double* u1, double* u2, double* u3, double* s1,
362  int ifPrintLog = VERBOSE_MODE) {
363  double *gp = (double*) _gp;
364 
365  // u1, u2, u3, sigma2 见113
366  double u10 = u0;
367  double u20 = u0;
368  double u30 = u0;
369  double s10 = s0; //*s1+1;
370 
371  // 收敛精度要求
372  int step = 1;
373  if (ifPrintLog) {
374  cout << "=======EMSteps======" << endl;
375  cout << "[step]\tu1\tu2\tu3\ts1\n";
376  }
377  while (fabs(u10 - *u1) > ACCURACY_REQ || fabs(u20 - *u2) > ACCURACY_REQ
378  || fabs(u30 - *u3) > ACCURACY_REQ || fabs(s10 - *s1) > ACCURACY_REQ) {
379  double py1 = 0, py10 = 0;
380  double py2 = 0, py20 = 0;
381  double py3 = 0, py30 = 0;
382  double py4 = 0, py5 = 0, py6 = 0;
383 
384  int i;
385  for (i = 0; i < sampleSize; i++) {
386  double gpi1f = *(gp + i * sampleSize + 0) * f(y[i], u10, s10);
387  double gpi2f = *(gp + i * sampleSize + 1) * f(y[i], u20, s10);
388  double gpi3f = *(gp + i * sampleSize + 2) * f(y[i], u30, s10);
389  double gpi = gpi1f + gpi2f + gpi3f;
390 
391  py1 += gpi1f / gpi; //PI1的值
392  py2 += gpi2f / gpi; //PI2的值
393  py3 += gpi3f / gpi; //PI3的值
394 
395  py10 += gpi1f / gpi * y[i]; //PI1和PI1*YI的和
396  py20 += gpi2f / gpi * y[i]; //PI2和PI2*YI的和
397  py30 += gpi3f / gpi * y[i]; //PI3和PI3*YI的和
398 
399  py4 += gpi1f / gpi * (y[i] - u10) * (y[i] - u10);
400  py5 += gpi2f / gpi * (y[i] - u20) * (y[i] - u20);
401  py6 += gpi3f / gpi * (y[i] - u30) * (y[i] - u30);
402  }
403 
404  *u1 = u10;
405  u10 = py10 / py1;
406  *u2 = u20;
407  u20 = py20 / py2;
408  *u3 = u30;
409  u30 = py30 / py3;
410  *s1 = s10;
411  s10 = (py4 + py5 + py6) / sampleSize;
412 
413  if (ifPrintLog) {
414  cout << "[" << step++ << "]\t" << *u1 << "\t" << *u2 << "\t" << *u3
415  << "\t" << *s1 << endl;
416  }
417  } //end-while
418 
419  //输出终值
420  *u1 = u10;
421  *u2 = u20;
422  *u3 = u30;
423  *s1 = s10;
424  if (ifPrintLog) {
425  cout << endl << "Finals:" << endl;
426  cout << "u1=" << *u1 << "\tu2=" << *u2 << "\tu3=" << *u3 << "\ts1="
427  << *s1 << endl;
428  }
429 }

499  EMCalculate(y, gp, sampleSize, u0, s0, &u1, &u2, &u3, &s1, ifPrintLog);
