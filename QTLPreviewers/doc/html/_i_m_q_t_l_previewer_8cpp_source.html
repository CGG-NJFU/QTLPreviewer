175 double calcLOD(const vector<vector<double> > gp, const vector<EXPData> expData,
176  const EXPData u0, const EXPData s0,
177  const EXPData u1, const EXPData u2, const EXPData u3, const EXPData s1 ) {
178  double sum1 = 0, sum2 = 0;
179 
180  unsigned int sampleSize = gp.size();
181  if (sampleSize != expData.size()) {
182  logger <<Priority::WARN <<"DATASET size error might be in calculateLOD function";
183  }
184 
185  logger <<Priority::DEBUG <<"start to calculate LOD";
186  logger <<Priority::DEBUG <<"u0=" <<u0 <<"\ts0=" <<s0;
187  for (unsigned int i = 0; i < sampleSize; i++) {
188  sum1 += log10(f(expData[i], u0, s0));
189  sum2 += log10(
190  gp[i][0] * f(expData[i], u1, s1)
191  + gp[i][1] * f(expData[i], u2, s1)
192  + gp[i][2] * f(expData[i], u3, s1));
193  }
194 
195  // LOD=log(L0/L)=logL0-LogL
196  // 连乘Log改为加法，除法改减法
197  double lod;
198  lod = sum2 - sum1;
199 
200  logger <<Priority::INFO <<"sum1=" <<sum1 <<"\tsum2=" <<sum2 <<"\tLOD=" <<lod;
201 
202  return lod;
203 }

216 void EMCalc(const vector<vector<double> >& gp, const vector<EXPData> expData,
217  const EXPData u0, const EXPData s0, EXPData* u1, EXPData* u2, EXPData* u3, EXPData* s1) {
218  unsigned int sampleSize = expData.size();
219  if (sampleSize != gp.size()) {
220  logger <<Priority::WARN <<"DATASET size error might be in EMCalculate function";
221  }
222 
224  EXPData u10 = u0;
225  EXPData u20 = u0;
226  EXPData u30 = u0;
227  EXPData s10 = s0; //*s1+1;
228 
230  int step = 1; //日志输出分步初试值为1
231 
232  logger <<Priority::DEBUG <<"=======EMSteps======";
233  logger <<Priority::DEBUG <<"[step]\tu1\tu2\tu3\ts1";
234 
235  while (fabs(u10 - *u1) > ACCURACY_REQ || fabs(u20 - *u2) > ACCURACY_REQ
236  || fabs(u30 - *u3) > ACCURACY_REQ || fabs(s10 - *s1) > ACCURACY_REQ) {
237  double py1 = 0, py10 = 0;
238  double py2 = 0, py20 = 0;
239  double py3 = 0, py30 = 0;
240  double py4 = 0, py5 = 0, py6 = 0;
241 
242  unsigned int i;
243  for (i = 0; i < sampleSize; i++) {
244  double gpi1f = gp[i][0] * f(expData[i], u10, s10);
245  double gpi2f = gp[i][1] * f(expData[i], u20, s10);
246  double gpi3f = gp[i][2] * f(expData[i], u30, s10);
247  double gpi = gpi1f + gpi2f + gpi3f;
248 
249  py1 += gpi1f / gpi; //PI1的值
250  py2 += gpi2f / gpi; //PI2的值
251  py3 += gpi3f / gpi; //PI3的值
252 
253  py10 += gpi1f / gpi * expData[i]; //PI1和PI1*YI的和
254  py20 += gpi2f / gpi * expData[i]; //PI2和PI2*YI的和
255  py30 += gpi3f / gpi * expData[i]; //PI3和PI3*YI的和
256 
257  py4 += gpi1f / gpi * (expData[i] - u10) * (expData[i] - u10);
258  py5 += gpi2f / gpi * (expData[i] - u20) * (expData[i] - u20);
259  py6 += gpi3f / gpi * (expData[i] - u30) * (expData[i] - u30);
260  }
261 
262  *u1 = u10;
263  u10 = py10 / py1;
264  *u2 = u20;
265  u20 = py20 / py2;
266  *u3 = u30;
267  u30 = py30 / py3;
268  *s1 = s10;
269  s10 = (py4 + py5 + py6) / sampleSize;
270 
271  logger <<Priority::DEBUG << "[" << step++ << "]\t" << *u1 << "\t" << *u2 << "\t" << *u3
272  << "\t" << *s1;
273  } //end-while
274 
275  //输出终值
276  *u1 = u10;
277  *u2 = u20;
278  *u3 = u30;
279  *s1 = s10;
280  logger <<Priority::INFO << "Result:\tu1=" <<*u1 <<"\tu2=" <<*u2 <<"\tu3=" <<*u3 <<"\ts1=" <<*s1;
281 }

292 double calcCross(const bitset<GENE_CP_BIT> geneBit,
293  const double r1, const double r2, const double r=0, const double zeta=1) {
294  if (geneBit.size()!=GENE_CP_BIT) return 0;
295 
296  double r12 = r1*r2*zeta;
297  double r0 = r;
298  if (r==0) {
299  r0 = r1+r2-2*r12;
300  }
301 
302  // 配子产生的概率，见书105页
303  switch(geneBit.to_ulong()) {
304  case 0: //M1QM2 000
305  case 7: //m1qm2 111
306  return (1.0-r1-r2+r12);// `=(1-r1)(1-r2);
307  case 1: //M1Qm2 001
308  case 6: //m1qM2 110
309  return (r2-r12);// `=r2(1-r1);
310  case 4: //M1qm2 100
311  case 3: //m1QM2 011
312  return (r1-r12);// `=r1(1-r2);
313  case 5: //M1qM2 101
314  case 2: //m1Qm2 010
315  return r12;// `=r1*r2;
316  }
317 
318  return 0;
319 }

356  logger <<Priority::WARN <<"DATASET size error might be in calcGeneCP function";
357  }
358 
359  logger <<Priority::DEBUG <<"start to calculate full Conditional Possibility for all Children";
360  for (unsigned int i=0; i<mSize; i++) {
361  bitset<4> bs_fm(i/GENE_CP_CUT); //前4位为亲本基因型
362  bitset<2> bs_qtl(i%GENE_CP_CUT); //后2位为QTL基因型
363 
364  int fGene = bs_fm[3]*4 + bs_qtl[1]*2 + bs_fm[1]; //父本基因型
365  int mGene = bs_fm[2]*4 + bs_qtl[0]*2 + bs_fm[0]; //母本基因型
366 
367  bitset<3> bs_up(fGene);
368  bitset<3> bs_down(mGene);
369 
370  rMatrix[i] = calcCross(bs_up, r1, r2, r, zeta) * calcCross(bs_down, r1, r2, r, zeta);
371 
372  geneMatrix[i] =
373  (i&0x20?f.substr(1,1):f.substr(0,1)) +
374  (i&0x10?f.substr(3,1):f.substr(2,1)) +
375  (i&0x08?m.substr(1,1):m.substr(0,1)) +
376  (i&0x04?m.substr(3,1):m.substr(2,1)) +
377  (i&0x02?q.substr(1,1):q.substr(0,1)) +
378  (i&0x01?q.substr(1,1):q.substr(0,1));
379  }
380 
381  if (ifReorder) {
382  sortGene(geneMatrix);
383  }
384
